defprocessor

macro

(defprocessor name xchg myfn)

Define as name a Camel Processor that runs fn on the Exchange in x.

;; upper-case
(defprocessor upperCaser xchg
  (let [body (.getBody (.getIn xchg))]
    (.setBody xchg (clojure.string/upper-case body))))

;; upper-case, reverse
(defprocessor revUpper x
  (->> (.getBody (.getIn x))
       clojure.string/upper-case
       clojure.string/reverse
       (.setBody xchg))
